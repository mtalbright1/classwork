class Keyboard {		
	static Array keyboard;

	function void init() {
        	let keyboard = 24576;		// RAM register 24576, is set to all 0 when no key is pressed
        	return;
    	} 

	function char keyPressed() {
		var int decimal, i, binary, power, divided;
		let decimal = 0;
		let i = 0;
		let binary = Memory.peek(24576);

		if (binary > 0) {
			while (i < 16) {
    				let divided = binary / power;
    				if ((Math.floor(divided / 2)) * 2 < Math.floor(divided)) {		// Odd check, only add if the binary digit is 1
        				let decimal = decimal + power;
    				}
    				power = power * 2; 		// Move to next power of 2
    				i = i + 1;
			}
			return decimal;		// I guess this int gets converted to a char?
    		}
		else {
        		return 0;
    		}
	}

	function char readChar() {
		var int key;
		let key = Keyboard.keyPressed();

		while (key = 0) {		// keep checking while there's nothing pressed
			let key = Keyboard.keyPressed();
		}
		while (key > 0) {		// wait until the key is released
			let key = Keyboard.keyPressed();
		}
		Output.printChar(key);
		return key;
	}

	function String readLine(String message) {
		var String s;
		var char c;
		let c = readChar();
		let s = String.new(message.length());

		while (true) {
			if (c = newLine) {
				do Output.println();
				return s;
			}
			else if (c = backSpace) {
				do s.eraseLastChar();
				do Output.backSpace();
			}
			else {
				do s.append(c);
			}
		}
		return s;
	}

	function int readInt(String message) {
		var String s;
		var char c;
		let c = readChar();
		let s = String.new(message.length());

		while (true) {
			if (c = 34) {		// 34 is newLine
				do Output.println();
				return s;
			}
			else if (c = 128) {		// 128 is backSpace
				do s.eraseLastChar();
				do Output.backSpace();
			}
			else {
				do s.append(c);
			}
		}
		return s.intValue();
	}
}
