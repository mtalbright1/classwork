class Sys {

	function void init() {
		do Math.init();
		do Memory.init();
		do Screen.init();
        	do Output.init();
        	do Keyboard.init();
        	do Main.main();
        	do Sys.halt();
        	return;
	}

	function void halt() {
		while (true) {
		}
		return;
	}

	function error(int errorCode) {		// prints the errorCode then halts
		do Output.printString("Error: ");
        	do Output.printInt(errCode);
        	do Sys.halt();
        	return;
	}

	function void wait(int duration) {
		var int i, j;
        	let i = 0;

        	while (i < duration) {
            		let j = 0;
            		while (j < 100) {
               			let j = j + 1;
            		}
            		let i = i + 1;
        	}
        	return;
	}
}
